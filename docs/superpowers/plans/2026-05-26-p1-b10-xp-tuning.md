Use existing tokens (`kCard`, `kBorder`, `kNeon`, `kMutedText`, `kCardRadius`, `kSpace*`) and `AppFonts.shareTechMono` / PressStart2P, matching surrounding cards.

```dart
class _XpBreakdownCard extends StatelessWidget {
  const _XpBreakdownCard({
    required this.baseXP,
    required this.lckMultiplier,
    required this.potionMultiplier,
    required this.totalXP,
  });

  final int baseXP;
  final double lckMultiplier;
  final double potionMultiplier;
  final int totalXP;

  @override
  Widget build(BuildContext context) {
    final rows = <Widget>[
      _row('+$baseXP', 'Workout complete', kNeon),
    ];
    if (lckMultiplier > 1.0) {
      rows.add(_row('×${lckMultiplier.toStringAsFixed(1)}', 'LCK multiplier', kAmber));
    }
    if (potionMultiplier > 1.0) {
      rows.add(_row('×${potionMultiplier.toStringAsFixed(1)}', 'XP potion', kCyan));
    }
    return Container(
      width: double.infinity,
      padding: const EdgeInsets.all(kCardPadding),
      decoration: BoxDecoration(
        color: kCard,
        border: Border.all(color: kBorder),
        borderRadius: BorderRadius.circular(kCardRadius),
      ),
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          const Text('SESSION XP',
              style: TextStyle(fontFamily: 'PressStart2P', fontSize: 10, color: kNeon)),
          const SizedBox(height: kSpace3),
          ...rows,
          const Divider(height: 24, color: kBorder),
          _row('+$totalXP', 'Total', kText),
        ],
      ),
    );
  }

  Widget _row(String amount, String label, Color amountColor) {
    return Padding(
      padding: const EdgeInsets.symmetric(vertical: 3),
      child: Row(
        children: [
          SizedBox(
            width: 56,
            child: Text(amount,
                style: AppFonts.shareTechMono(color: amountColor, fontSize: 14, fontWeight: FontWeight.w700)),
          ),
          const SizedBox(width: kSpace3),
          Expanded(
            child: Text(label, style: AppFonts.shareTechMono(color: kMutedText, fontSize: 13)),
          ),
        ],
      ),
    );
  }
}
```
